234 /**
235  * EM pool configuration
236  *
237  * Configuration of an EM event pool consisting of up to 'EM_MAX_SUBPOOLS'
238  * subpools, each supporting a specific event payload size. Event allocation,
239  * i.e. em_alloc(), will use the subpool that provides the best fit for the
240  * requested size.
241  */
242 typedef struct {
243  /**
244  * Event type determines the pool type used:
245  * - EM_EVENT_TYPE_SW creates subpools of type 'ODP_POOL_BUFFER'
246  * This kind of EM pool CANNOT be used to create events of major
247  * type EM_EVENT_TYPE_PACKET.
248  * - EM_EVENT_TYPE_PACKET creates subpools of type 'ODP_POOL_PACKET'
249  * This kind of EM pool can be used for events of all kinds.
250  * @note Only major types are considered here, setting minor is error
251  */
252  em_event_type_t event_type;
253  /**
254  * Alignment offset in bytes for the event payload start address
255  * (for all events allocated from this EM pool).
256  *
257  * The default EM event payload start address alignment is a
258  * power-of-two that is at minimum 32 bytes (i.e. 32 B, 64 B, 128 B etc.
259  * depending on e.g. target cache-line size).
260  * The 'align_offset.value' option can be used to fine-tune the
261  * start-address by a small offset to e.g. make room for a small
262  * SW header before the rest of the payload that might need a specific
263  * alignment for direct HW-access.
264  * Example: setting 'align_offset.value = 8' makes sure that the payload
265  * _after_ 8 bytes will be aligned at minimum (2^x) 32 bytes.
266  *
267  * This option conserns all events allocated from the pool and overrides
268  * the global config file option 'pool.align_offset' for this pool.
269  */
270  struct {
271  /**
272  * Select: Use pool-specific align-offset 'value' from below or
273  * use the global default value from the config file.
274  * false (0): Use default value from the config file.
275  * true (not 0): Use pool specific value set below.
276  */
277  int in_use;
278  /**
279  * Pool-specific event payload alignment offset value in bytes
280  * (only evaluated if 'in_use=true').
281  * Overrides the config file value for this pool.
282  * The given 'value' must be a small power-of-two: 2, 4, or 8
283  * 0: Explicitly set 'No align offset' for the pool.
284  */
285  uint32_t value;
286  } align_offset;

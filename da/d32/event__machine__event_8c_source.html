33 em_event_t
34 em_alloc(size_t size, em_event_type_t type, em_pool_t pool)
35 {
36  mpool_elem_t *pool_elem;
37  em_event_t event;
38 
39  pool_elem = pool_elem_get(pool);
40  if (unlikely(size == 0 ||
41  pool_elem == NULL || !pool_allocated(pool_elem))) {
42  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_ALLOC,
43  "Invalid args: size:%zu type:%u pool:%" PRI_POOL "",
44  size, type, pool);
45  return EM_EVENT_UNDEF;
46  }
47 
48  if (pool_elem->event_type == EM_EVENT_TYPE_PACKET) {
49  /*
50  * EM event pools created with type=PKT can support SW events
51  * as well as pkt events.
52  */
53  event = event_alloc_pkt(pool_elem, size, type);
54  } else { /* pool_elem->event_type == EM_EVENT_TYPE_SW */
55  /*
56  * EM event pools created with type=SW can not support
57  * pkt events.
58  */
59  if (unlikely(em_get_type_major(type) ==
60  EM_EVENT_TYPE_PACKET)) {
61  INTERNAL_ERROR(EM_ERR_NOT_IMPLEMENTED, EM_ESCOPE_ALLOC,
62  "EM-pool:%s(%" PRI_POOL "):\n"
63  "Invalid event type:%u for buf",
64  pool_elem->name, pool, type);
65  return EM_EVENT_UNDEF;
66  }
67 
68  event = event_alloc_buf(pool_elem, size, type);
69  }
70 
71  if (unlikely(event == EM_EVENT_UNDEF)) {
72  em_status_t err =
73  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_ALLOC,
74  "EM-pool:'%s': sz:%zu type:0x%x pool:%" PRI_POOL "",
75  pool_elem->name, size, type, pool);
76  if (EM_CHECK_LEVEL > 1 && err != EM_OK &&
77  em_shm->opt.pool.statistics_enable) {
78  em_pool_info_print(pool);
79  }
80  return EM_EVENT_UNDEF;
81  }
82 
83  if (EM_API_HOOKS_ENABLE)
84  call_api_hooks_alloc(event, size, type, pool);
85 
86  return event;
87 }

89 void
90 em_free(em_event_t event)
91 {
92  odp_event_t odp_event;
93 
94  if (unlikely(event == EM_EVENT_UNDEF)) {
95  INTERNAL_ERROR(EM_ERR_BAD_POINTER, EM_ESCOPE_FREE,
96  "event undefined!");
97  return;
98  }
99 
100  if (EM_CHECK_LEVEL > 1 || em_shm->opt.pool.statistics_enable) {
101  event_hdr_t *const ev_hdr = event_to_hdr(event);
102 
103  if (EM_CHECK_LEVEL > 1) {
104  /* Simple double-free detection */
105  const uint32_t allocated =
106  env_atomic32_sub_return(&ev_hdr->allocated, 1);
107 
108  if (unlikely(allocated != 0)) {
109  const char *const fmt =
110  "Double free:event:%" PRI_EVENT "";
111  INTERNAL_ERROR(EM_FATAL(EM_ERR_BAD_POINTER),
112  EM_ESCOPE_FREE, fmt, event);
113  }
114  }
115  if (em_shm->opt.pool.statistics_enable) {
116  /* Update pool statistcs */
117  em_pool_t pool = ev_hdr->pool;
118 
119  if (pool != EM_POOL_UNDEF) {
120  mpool_elem_t *pelem = pool_elem_get(pool);
121  int subpool = ev_hdr->subpool;
122 
123  if (pelem)
124  pool_stat_decrement(pool, subpool);
125  }
126  }
127  }
128 
129  if (EM_API_HOOKS_ENABLE)
130  call_api_hooks_free(event);
131 
132  odp_event = event_em2odp(event);
133  odp_event_free(odp_event);
134 }

136 em_status_t
137 em_send(em_event_t event, em_queue_t queue)
138 {
139  const int is_external = queue_external(queue);
140  queue_elem_t *q_elem = NULL;
141  event_hdr_t *ev_hdr;
142  int num_sent;
143  em_status_t stat;
144 
145  /*
146  * Check all args.
147  */
148  if (EM_CHECK_LEVEL > 0)
149  RETURN_ERROR_IF(event == EM_EVENT_UNDEF,
150  EM_ERR_BAD_ID, EM_ESCOPE_SEND,
151  "Invalid event");
152 
153  ev_hdr = event_to_hdr(event);
154  ev_hdr->egrp = EM_EVENT_GROUP_UNDEF;
155 
156  if (EM_CHECK_LEVEL > 2)
157  RETURN_ERROR_IF(env_atomic32_get(&ev_hdr->allocated) != 1,
158  EM_FATAL(EM_ERR_BAD_STATE), EM_ESCOPE_SEND,
159  "Event:%" PRI_EVENT " already freed!", event);
160 
161  if (!is_external) {
162  /* queue belongs to this EM instance */
163  q_elem = queue_elem_get(queue);
164  if (EM_CHECK_LEVEL > 0)
165  RETURN_ERROR_IF(q_elem == NULL,
166  EM_ERR_BAD_ID, EM_ESCOPE_SEND,
167  "Invalid queue:%" PRI_QUEUE "", queue);
168  if (EM_CHECK_LEVEL > 1)
169  RETURN_ERROR_IF(!queue_allocated(q_elem),
170  EM_ERR_BAD_STATE, EM_ESCOPE_SEND,
171  "Invalid queue:%" PRI_QUEUE "", queue);
172  }
173 
174  /* Buffer events from EO-start sent to scheduled queues */
175  if (unlikely(em_locm.start_eo_elem != NULL &&
176  !is_external && q_elem->scheduled)) {
177  /*
178  * em_send() called from within an EO-start function:
179  * all events sent to scheduled queues will be buffered
180  * and sent when the EO-start operation completes.
181  */
182  num_sent = eo_start_buffer_events(&event, 1, queue,
183  EM_EVENT_GROUP_UNDEF);
184  stat = num_sent == 1 ? EM_OK : EM_ERR_OPERATION_FAILED;
185  if (EM_CHECK_LEVEL == 0)
186  return stat;
187  RETURN_ERROR_IF(stat != EM_OK, stat, EM_ESCOPE_SEND,
188  "send from EO-start failed");
189  return EM_OK;
190  }
191 
192  if (EM_API_HOOKS_ENABLE)
193  call_api_hooks_send(&event, 1, queue, EM_EVENT_GROUP_UNDEF);
194 
195  if (is_external) {
196  /*
197  * Send out of EM to another device via event-chaining and a
198  * user-provided function 'event_send_device()'
199  */
200  stat = send_chaining(event, ev_hdr, queue);
201  if (EM_CHECK_LEVEL == 0)
202  return stat;
203  RETURN_ERROR_IF(stat != EM_OK, stat, EM_ESCOPE_SEND,
204  "send_chaining failed: Q:%" PRI_QUEUE "",
205  queue);
206  return EM_OK;
207  }
208 
209  /*
210  * Normal send to a queue on this device
211  */
212  switch (q_elem->type) {
213  case EM_QUEUE_TYPE_ATOMIC:
214  case EM_QUEUE_TYPE_PARALLEL:
215  case EM_QUEUE_TYPE_PARALLEL_ORDERED:
216  stat = send_event(event, q_elem);
217  break;
218  case EM_QUEUE_TYPE_UNSCHEDULED:
219  stat = queue_unsched_enqueue(event, q_elem);
220  break;
221  case EM_QUEUE_TYPE_LOCAL:
222  stat = send_local(event, ev_hdr, q_elem);
223  break;
224  case EM_QUEUE_TYPE_OUTPUT:
225  stat = send_output(event, q_elem);
226  break;
227  default:
228  stat = EM_ERR_NOT_FOUND;
229  break;
230  }
231 
232  if (EM_CHECK_LEVEL == 0)
233  return stat;
234  RETURN_ERROR_IF(stat != EM_OK, stat, EM_ESCOPE_SEND,
235  "send failed: Q:%" PRI_QUEUE " type:%" PRI_QTYPE "",
236  queue, q_elem->type);
237 
238  return EM_OK;
239 }

241 int
242 em_send_multi(em_event_t *const events, int num, em_queue_t queue)
243 {
244  event_hdr_t *ev_hdrs[num];
245  const int is_external = queue_external(queue);
246  queue_elem_t *q_elem = NULL;
247  int num_sent;
248  int i;
249 
250  /*
251  * Check all args.
252  */
253  if (EM_CHECK_LEVEL > 0) {
254  if (unlikely(events == NULL || num <= 0)) {
255  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_SEND_MULTI,
256  "Invalid events");
257  return 0;
258  }
259  }
260  if (EM_CHECK_LEVEL > 2) {
261  for (i = 0; i < num && events[i] != EM_EVENT_UNDEF; i++)
262  ;
263  if (unlikely(i != num)) {
264  INTERNAL_ERROR(EM_ERR_BAD_POINTER,
265  EM_ESCOPE_SEND_MULTI,
266  "Invalid events[%d]=%" PRI_EVENT "",
267  i, events[i]);
268  return 0;
269  }
270  }
271 
272  event_to_hdr_multi(events, ev_hdrs, num);
273  for (i = 0; i < num; i++)
274  ev_hdrs[i]->egrp = EM_EVENT_GROUP_UNDEF;
275 
276  if (EM_CHECK_LEVEL > 2) {
277  for (i = 0; i < num &&
278  env_atomic32_get(&ev_hdrs[i]->allocated) == 1; i++)
279  ;
280  if (unlikely(i != num)) {
281  const char *const fmt =
282  "events[%d]:%" PRI_EVENT " already freed!";
283  INTERNAL_ERROR(EM_FATAL(EM_ERR_BAD_STATE),
284  EM_ESCOPE_SEND_MULTI, fmt, i, events[i]);
285  return 0;
286  }
287  }
288 
289  if (!is_external) {
290  /* queue belongs to this EM instance */
291  q_elem = queue_elem_get(queue);
292  if (EM_CHECK_LEVEL > 0) {
293  if (unlikely(q_elem == NULL)) {
294  INTERNAL_ERROR(EM_ERR_BAD_ID,
295  EM_ESCOPE_SEND_MULTI,
296  "Invalid queue:%" PRI_QUEUE "",
297  queue);
298  return 0;
299  }
300  }
301  if (EM_CHECK_LEVEL > 1) {
302  if (unlikely(!queue_allocated(q_elem))) {
303  INTERNAL_ERROR(EM_ERR_BAD_STATE,
304  EM_ESCOPE_SEND_MULTI,
305  "Invalid queue:%" PRI_QUEUE "",
306  queue);
307  return 0;
308  }
309  }
310  }
311 
312  /* Buffer events from EO-start sent to scheduled queues */
313  if (unlikely(em_locm.start_eo_elem != NULL &&
314  !is_external && q_elem->scheduled)) {
315  /*
316  * em_send_multi() called from within an EO-start function:
317  * all events sent to scheduled queues will be buffered
318  * and sent when the EO-start operation completes.
319  */
320  num_sent = eo_start_buffer_events(events, num, queue,
321  EM_EVENT_GROUP_UNDEF);
322  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num))
323  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_SEND_MULTI,
324  "send-multi EO-start: req:%d, sent:%d",
325  num, num_sent);
326  return num_sent;
327  }
328 
329  if (EM_API_HOOKS_ENABLE)
330  call_api_hooks_send(events, num, queue, EM_EVENT_GROUP_UNDEF);
331 
332  if (is_external) {
333  /*
334  * Send out of EM to another device via event-chaining and a
335  * user-provided function 'event_send_device_multi()'
336  */
337  num_sent = send_chaining_multi(events, ev_hdrs, num, queue);
338  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num)) {
339  INTERNAL_ERROR(EM_ERR_OPERATION_FAILED,
340  EM_ESCOPE_SEND_MULTI,
341  "send-egrp-multi fails: req:%d, sent:%d",
342  num, num_sent);
343  }
344  return num_sent;
345  }
346 
347  /*
348  * Normal send to a queue on this device
349  */
350  switch (q_elem->type) {
351  case EM_QUEUE_TYPE_ATOMIC:
352  case EM_QUEUE_TYPE_PARALLEL:
353  case EM_QUEUE_TYPE_PARALLEL_ORDERED:
354  num_sent = send_event_multi(events, num, q_elem);
355  break;
356  case EM_QUEUE_TYPE_UNSCHEDULED:
357  num_sent = queue_unsched_enqueue_multi(events, num, q_elem);
358  break;
359  case EM_QUEUE_TYPE_LOCAL:
360  num_sent = send_local_multi(events, ev_hdrs, num, q_elem);
361  break;
362  case EM_QUEUE_TYPE_OUTPUT:
363  num_sent = send_output_multi(events, num, q_elem);
364  break;
365  default:
366  num_sent = 0;
367  break;
368  }
369 
370  if (EM_CHECK_LEVEL > 0 && unlikely(num_sent != num)) {
371  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_SEND_MULTI,
372  "send-multi failed: req:%d, sent:%d",
373  num, num_sent);
374  }
375 
376  return num_sent;
377 }

379 void *
380 em_event_pointer(em_event_t event)
381 {
382  odp_event_t odp_event;
383  odp_event_type_t odp_etype;
384 
385  if (unlikely(event == EM_EVENT_UNDEF)) {
386  INTERNAL_ERROR(EM_ERR_BAD_POINTER, EM_ESCOPE_EVENT_POINTER,
387  "event undefined!");
388  return NULL;
389  }
390 
391  odp_event = event_em2odp(event);
392  odp_etype = odp_event_type(odp_event);
393 
394  switch (odp_etype) {
395  case ODP_EVENT_PACKET: {
396  odp_packet_t odp_pkt = odp_packet_from_event(odp_event);
397 
398  return odp_packet_data(odp_pkt);
399  }
400  case ODP_EVENT_BUFFER: {
401  odp_buffer_t odp_buf = odp_buffer_from_event(odp_event);
402  event_hdr_t *const ev_hdr = odp_buffer_addr(odp_buf);
403 
404  return (void *)((uintptr_t)ev_hdr + sizeof(event_hdr_t)
405  - ev_hdr->align_offset);
406  }
407  case ODP_EVENT_TIMEOUT: /* @TBD */
408  case ODP_EVENT_CRYPTO_COMPL: /* @TBD */
409  default:
410  INTERNAL_ERROR(EM_ERR_NOT_IMPLEMENTED, EM_ESCOPE_EVENT_POINTER,
411  "Unexpected odp event type:%u", odp_etype);
412  return NULL;
413  }
414 }

39 /*
40  * Max supported value for the config file option 'pool.align_offset'.
41  *
42  * The limitation is set by events based on odp-bufs that include the ev-hdr at
43  * the beginning of the odp-buf payload - the alignment is adjusted into the end
44  * of the ev-hdr.
45  * Events based on odp-pkts do not have this restriction but the same limit is
46  * used for all.
47  */
48 #define ALIGN_OFFSET_MAX ((int)(sizeof(event_hdr_t) - \
49  offsetof(event_hdr_t, end_hdr_data)))

58 static int
59 read_config_file(void)
60 {
61  const char *conf_str;
62  bool val_bool = false;
63  int val = 0;
64  int ret;
65 
66  EM_PRINT("EM-pool config:\n");
67 
68  /*
69  * Option: pool.statistics_enable
70  */
71  conf_str = "pool.statistics_enable";
72  ret = libconfig_lookup_bool(&em_shm->libconfig, conf_str, &val_bool);
73  if (unlikely(!ret)) {
74  EM_LOG(EM_LOG_ERR, "Config option '%s' not found", conf_str);
75  return -1;
76  }
77  /* store & print the value */
78  em_shm->opt.pool.statistics_enable = (int)val_bool;
79  EM_PRINT(" %s: %s(%d)\n", conf_str, val_bool ? "true" : "false",
80  val_bool);
81 
82  /*
83  * Option: pool.align_offset
84  */
85  conf_str = "pool.align_offset";
86  ret = libconfig_lookup_int(&em_shm->libconfig, conf_str, &val);
87  if (unlikely(!ret)) {
88  EM_LOG(EM_LOG_ERR, "Config option '%s' not found.\n", conf_str);
89  return -1;
90  }
91  if (val < 0 || val > ALIGN_OFFSET_MAX || !POWEROF2(val)) {
92  EM_LOG(EM_LOG_ERR, "Bad config value '%s = %d'\n",
93  conf_str, val);
94  return -1;
95  }
96  /* store & print the value */
97  em_shm->opt.pool.align_offset = val;
98  EM_PRINT(" %s: %d (max: %d)\n",
99  conf_str, val, ALIGN_OFFSET_MAX);
100 
101  return 0;
102 }

104 em_status_t
105 pool_init(mpool_tbl_t *const mpool_tbl, mpool_pool_t *const mpool_pool,
106  em_pool_cfg_t *const default_pool_cfg)
107 {
108  em_pool_t pool;
109  int i, j, ret;
110  const int cores = em_core_count();
111 
112  memset(mpool_tbl, 0, sizeof(mpool_tbl_t));
113  memset(mpool_pool, 0, sizeof(mpool_pool_t));
114  env_atomic32_init(&em_shm->pool_count);
115 
116  ret = objpool_init(&mpool_pool->objpool, cores);
117  if (ret != 0)
118  return EM_ERR_OPERATION_FAILED;
119 
120  for (i = 0; i < EM_CONFIG_POOLS; i++) {
121  em_pool_t pool = pool_idx2hdl(i);
122  mpool_elem_t *mpool_elem = pool_elem_get(pool);
123 
124  mpool_elem->em_pool = pool;
125  mpool_elem->event_type = EM_EVENT_TYPE_UNDEF;
126  for (j = 0; j < EM_MAX_SUBPOOLS; j++) {
127  mpool_elem->odp_pool[j] = ODP_POOL_INVALID;
128  mpool_elem->size[j] = 0;
129  }
130 
131  objpool_add(&mpool_pool->objpool, i % cores,
132  &mpool_elem->objpool_elem);
133  }
134 
135  if (read_config_file())
136  return EM_ERR_LIB_FAILED;
137 
138  /* Store common ODP pool capabilities in the mpool_tbl for easy access*/
139  if (odp_pool_capability(&mpool_tbl->odp_pool_capability) != 0)
140  return EM_ERR_LIB_FAILED;
141 
142  /* Create the 'EM_POOL_DEFAULT' pool */
143  pool = em_pool_create(EM_POOL_DEFAULT_NAME, EM_POOL_DEFAULT,
144  default_pool_cfg);
145  if (pool == EM_POOL_UNDEF || pool != EM_POOL_DEFAULT)
146  return EM_ERR_ALLOC_FAILED;
147 
148  return EM_OK;
149 }

224 static int
225 invalid_pool_cfg(em_pool_cfg_t *const pool_cfg)
226 {
227  if (unlikely(pool_cfg == NULL ||
228  pool_cfg->num_subpools <= 0 ||
229  pool_cfg->num_subpools > EM_MAX_SUBPOOLS ||
230  (pool_cfg->event_type != EM_EVENT_TYPE_SW &&
231  pool_cfg->event_type != EM_EVENT_TYPE_PACKET)))
232  return -1;
233 
234  if (pool_cfg->align_offset.in_use) {
235  if (pool_cfg->align_offset.value > ALIGN_OFFSET_MAX ||
236  !POWEROF2(pool_cfg->align_offset.value))
237  return -2;
238  }
239 
240  for (int i = 0; i < pool_cfg->num_subpools; i++) {
241  if (unlikely(pool_cfg->subpool[i].size <= 0 ||
242  pool_cfg->subpool[i].num <= 0))
243  return -3;
244  }
245 
246  return 0;
247 }

249 em_pool_t
250 pool_create(const char *name, em_pool_t pool, em_pool_cfg_t *const pool_cfg)
251 {
252  em_pool_t allocated_pool;
253  em_pool_cfg_t sorted_cfg;
254  odp_pool_param_t pool_params;
255  odp_pool_t odp_pool;
256  char pool_name[ODP_POOL_NAME_LEN];
257  int i, j, n;
258  uint32_t size, num;
259  mpool_elem_t *mpool_elem;
260  odp_pool_capability_t *const pool_capa =
261  &em_shm->mpool_tbl.odp_pool_capability;
262 
263  /* Verify config */
264  if (unlikely(invalid_pool_cfg(pool_cfg)))
265  return EM_POOL_UNDEF;
266 
267  /* Allocate a free EM pool */
268  allocated_pool = pool_alloc(pool /* requested pool or 'undef'*/);
269  if (unlikely(allocated_pool == EM_POOL_UNDEF))
270  return EM_POOL_UNDEF;
271  pool = allocated_pool;
272 
273  mpool_elem = pool_elem_get(pool);
274  /* Sanity check */
275  if (mpool_elem->em_pool != pool)
276  return EM_POOL_UNDEF;
277 
278  mpool_elem->event_type = pool_cfg->event_type;
279  /* Store successfully created subpools later */
280  mpool_elem->num_subpools = 0;
281  /* Store the event pool name, if given */
282  if (name && *name) {
283  strncpy(mpool_elem->name, name, sizeof(mpool_elem->name));
284  mpool_elem->name[sizeof(mpool_elem->name) - 1] = '\0';
285  } else {
286  mpool_elem->name[0] = '\0';
287  }
288 
289  sorted_cfg = *pool_cfg;
290  /* Sort the subpools in ascending order based on the buffer size */
291  n = sorted_cfg.num_subpools;
292  for (i = 0; i < n; i++) {
293  for (j = i + 1; j < n; j++) {
294  if (sorted_cfg.subpool[i].size >
295  sorted_cfg.subpool[j].size) {
296  size = sorted_cfg.subpool[i].size;
297  num = sorted_cfg.subpool[i].num;
298  sorted_cfg.subpool[i] = sorted_cfg.subpool[j];
299  sorted_cfg.subpool[j].size = size;
300  sorted_cfg.subpool[j].num = num;
301  }
302  }
303  }
304  /* store the sorted config */
305  mpool_elem->pool_cfg = sorted_cfg;
306 
307  const int cores = em_core_count();
308  const int pool_idx = pool_hdl2idx(pool);
309  mpool_statistics_t *const pstat_core = em_shm->mpool_tbl.pool_stat_core;
310 
311  for (i = 0; i < cores; i++) {
312  for (j = 0; j < EM_MAX_SUBPOOLS; j++) {
313  pstat_core[i].stat[pool_idx][j].alloc = 0;
314  pstat_core[i].stat[pool_idx][j].free = 0;
315  }
316  }
317 
318  /*
319  * Align the event payload by pushing the start-location into the buffer
320  * 'headroom' - if done, then the subpool size can also be decremented
321  * by the same amount.
322  */
323  uint32_t align_offset = em_shm->opt.pool.align_offset;
324 
325  /* Pool-specific config overrides config file 'align_offset' value */
326  if (pool_cfg->align_offset.in_use)
327  align_offset = pool_cfg->align_offset.value;
328  /* Set subpool minimum alignment */
329  uint32_t odp_align = ODP_CACHE_LINE_SIZE;
330 
331  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
332  if (odp_align > pool_capa->pkt.max_align)
333  odp_align = pool_capa->pkt.max_align;
334  } else {
335  if (odp_align > pool_capa->buf.max_align)
336  odp_align = pool_capa->buf.max_align;
337  }
338  /* verify alignment requirements */
339  if (!POWEROF2(odp_align) || odp_align <= align_offset) {
340  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_POOL_CREATE,
341  "EM-pool:\"%s\" align mismatch:\n"
342  "align:%u cfg:align_offset:%u",
343  name, odp_align, align_offset);
344  goto error;
345  }
346  /* Store the event payload alignment requirement for the pool */
347  mpool_elem->align_offset = align_offset;
348 
349  for (i = 0; i < sorted_cfg.num_subpools; i++) {
350  odp_pool_param_init(&pool_params);
351  size = sorted_cfg.subpool[i].size;
352 
353  if (pool_cfg->event_type == EM_EVENT_TYPE_PACKET) {
354  pool_params.type = ODP_POOL_PACKET;
355  /* num == max_num */
356  pool_params.pkt.num = sorted_cfg.subpool[i].num;
357  pool_params.pkt.max_num = sorted_cfg.subpool[i].num;
358 
359  if (size > align_offset)
360  size = size - align_offset;
361  else
362  size = 1; /* 0:default, can be big => use 1 */
363  /* len == max_len */
364  pool_params.pkt.len = size;
365  pool_params.pkt.max_len = size;
366  pool_params.pkt.seg_len = size;
367  pool_params.pkt.align = odp_align;
368  /*
369  * Reserve space for the event header in each packet's
370  * user area:
371  */
372  pool_params.pkt.uarea_size = sizeof(event_hdr_t);
373  /*
374  * Reserve space for alloc-alignment in the headroom,
375  * use default headroom value unless it's smaller than
376  * needed:
377  */
378  if (pool_params.pkt.headroom < align_offset)
379  pool_params.pkt.headroom = align_offset;
380  } else { /* pool_cfg->event_type == EM_EVENT_TYPE_SW */
381  pool_params.type = ODP_POOL_BUFFER;
382  pool_params.buf.num = sorted_cfg.subpool[i].num;
383  pool_params.buf.size = size + sizeof(event_hdr_t)
384  - align_offset;
385  pool_params.buf.align = odp_align;
386  }
387 
388  snprintf(pool_name, sizeof(pool_name),
389  "%" PRI_POOL ":%d-%s", pool, i, mpool_elem->name);
390  pool_name[sizeof(pool_name) - 1] = '\0';
391 
392  odp_pool = odp_pool_create(pool_name, &pool_params);
393  if (unlikely(odp_pool == ODP_POOL_INVALID)) {
394  INTERNAL_ERROR(EM_FATAL(EM_ERR_ALLOC_FAILED),
395  EM_ESCOPE_POOL_CREATE,
396  "EM-subpool:\"%s\" create fails",
397  pool_name);
398  goto error;
399  }
400 
401  /*odp_pool_print(odp_pool);*/
402 
403  mpool_elem->odp_pool[i] = odp_pool;
404  mpool_elem->size[i] = sorted_cfg.subpool[i].size;
405  mpool_elem->num_subpools++; /* created subpools for delete */
406  }
407 
408  /* Success! */
409  return mpool_elem->em_pool;
410 
411 error:
412  (void)pool_delete(pool);
413  return EM_POOL_DEFAULT;
414 }

416 em_status_t
417 pool_delete(em_pool_t pool)
418 {
419  mpool_elem_t *const mpool_elem = pool_elem_get(pool);
420  int i;
421 
422  if (unlikely(mpool_elem == NULL || !pool_allocated(mpool_elem)))
423  return EM_ERR_BAD_ID;
424 
425  for (i = 0; i < mpool_elem->num_subpools; i++) {
426  odp_pool_t odp_pool = mpool_elem->odp_pool[i];
427  int ret;
428 
429  if (odp_pool == ODP_POOL_INVALID)
430  return EM_ERR_NOT_FOUND;
431 
432  ret = odp_pool_destroy(odp_pool);
433  if (unlikely(ret))
434  return EM_ERR_LIB_FAILED;
435 
436  mpool_elem->odp_pool[i] = ODP_POOL_INVALID;
437  mpool_elem->size[i] = 0;
438  }
439 
440  mpool_elem->name[0] = '\0';
441  mpool_elem->event_type = EM_EVENT_TYPE_UNDEF;
442  mpool_elem->num_subpools = 0;
443 
444  return pool_free(pool);
445 }

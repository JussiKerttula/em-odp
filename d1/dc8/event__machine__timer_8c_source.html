31  * Some notes about the implementation:
32  *
33  * EM timer add-on API is close to ODP timer, but there are issues
34  * making this code a bit more complex than it could be:
35  *
36  * 1) no periodic timer in ODP
37  * 2) unless using the pre-defined timeout event there is no way to access
38  * all necessary information runtime to implement a periodic timer
39  *
40  * Point 2 is solved by creating a timeout pool. When user allocates
41  * EM timeout, a new minimum size buffer is allocated to store all the needed
42  * information. Timer handle is a pointer to such buffer so all data is
43  * available via the handle (ack() is the most problematic case). This does
44  * create performance penalty, but so far it looks like the penalty is not
45  * too large and does simplify the code otherwise. Also timeouts could be
46  * pre-allocated as the API separates creation and arming.
47  * Most of the syncronization is handled by ODP timer, a ticketlock is used
48  * for high level management API.
49  *
50  */

56 #define TMR_I2H(x) ((em_timer_t)(uintptr_t)((x) + 1))
57 #define TMR_H2I(x) ((int)((uintptr_t)(x) - 1))

59 static int is_timer_valid(em_timer_t tmr)
60 {
61  unsigned int i;
62  timer_storage_t *const tmrs = &em_shm->timers;
63 
64  if (tmr == EM_TIMER_UNDEF)
65  return 0;
66  i = (unsigned int)TMR_H2I(tmr);
67  if (i >= EM_ODP_MAX_TIMERS)
68  return 0;
69 
70  if (tmrs->timer[i].odp_tmr_pool == ODP_TIMER_POOL_INVALID ||
71  tmrs->timer[i].tmo_pool == ODP_POOL_INVALID)
72  return 0;
73  return 1;
74 }

76 em_timer_t em_timer_create(const em_timer_attr_t *tmr_attr)
77 {
78  em_timer_attr_t attr = {.flags = EM_TIMER_FLAG_DEFAULT,
79  .clk_src = EM_TIMER_CLKSRC_DEFAULT};
80 
81  /* use provided timer-attr if given */
82  if (tmr_attr != NULL)
83  attr = *tmr_attr; /* copy */
84 
85  odp_timer_capability_t odp_capa;
86  odp_timer_clk_src_t odp_clksrc;
87  int err;
88 
89  memset(&odp_capa, 0, sizeof(odp_timer_capability_t));
90 
91  err = timer_clksrc_em2odp(attr.clk_src, &odp_clksrc);
92  if (err) {
93  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TIMER_CREATE,
94  "Unsupported EM-timer clock source:%d",
95  attr.clk_src);
96  return EM_TIMER_UNDEF;
97  }
98 
99  err = odp_timer_capability(odp_clksrc, &odp_capa);
100  if (err) {
101  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TIMER_CREATE,
102  "Timer capability:%d failed, odp-clksrc:%d",
103  err, odp_clksrc);
104  return EM_TIMER_UNDEF;
105  }
106 
107  /*
108  * Determine attr.num_tmo
109  */
110 
111  /* Account for pool buf-stashing on each core, assume max512 per core */
112  const uint32_t default_num_tmo = em_core_count() * 512 + 1024 - 1;
113 
114  if (odp_capa.max_timers == 0) {
115  /* 0 = limited only by the available pool memory size */
116  if (attr.num_tmo == 0)
117  attr.num_tmo = default_num_tmo;
118  /* else use attr.num_tmo as given */
119  } else {
120  /* odp has a limit for the number of odp-timers */
121  if (attr.num_tmo == 0) {
122  attr.num_tmo = MIN(odp_capa.max_timers,
123  default_num_tmo);
124  } else if (attr.num_tmo > odp_capa.max_timers) {
125  INTERNAL_ERROR(EM_ERR_TOO_LARGE, EM_ESCOPE_TIMER_CREATE,
126  "Timer capability err, odp-clksrc:%d\n"
127  " req num_tmo > odp-capa max_timers:\n"
128  " %" PRIu64 " > %" PRIu64 "",
129  odp_clksrc,
130  attr.num_tmo, odp_capa.max_timers);
131  return EM_TIMER_UNDEF;
132  }
133  }
134  /*
135  * Account for the thread local buffer cache used by the the odp-pools:
136  * a core should be able to create attr.num_tmo number of tmos even if
137  * some buffers are cached in other cores.
138  * Here assume the local cache size is max 256.
139  */
140  attr.num_tmo += (em_core_count() - 1) * 256;
141  /* don't increase above odp limit (if any) */
142  if (odp_capa.max_timers > 0 && attr.num_tmo > odp_capa.max_timers)
143  attr.num_tmo = odp_capa.max_timers;
144 
145  /*
146  * Determine attr.resolution and attr.max_tmo
147  */
148  uint64_t default_res =
149  odp_capa.max_res.res_ns > EM_ODP_TIMER_RESOL_DEF_NS ?
150  odp_capa.max_res.res_ns : EM_ODP_TIMER_RESOL_DEF_NS;
151  uint64_t min_tmo = 0;
152  odp_timer_res_capability_t odp_res_capa;
153 
154  memset(&odp_res_capa, 0, sizeof(odp_timer_res_capability_t));
155 
156  if (attr.resolution == 0) {
157  if (attr.max_tmo == 0)
158  odp_res_capa.res_ns = default_res;
159  else
160  odp_res_capa.max_tmo = attr.max_tmo;
161  } else {
162  odp_res_capa.res_ns = attr.resolution; /*verify usr resolution*/
163  }
164  err = odp_timer_res_capability(odp_clksrc, &odp_res_capa);
165  /* verify if user provided both attr.resolution and attr.max_tmo */
166  if (!err && attr.resolution > 0 && attr.max_tmo > 0 &&
167  odp_res_capa.max_tmo < attr.max_tmo)
168  err = 1; /* combination attr.resolution and .max_tmo invalid */
169 
170  if (err) {
171  INTERNAL_ERROR(EM_ERR_OPERATION_FAILED,
172  EM_ESCOPE_TIMER_CREATE,
173  "Timer resolution-capability:%d odp-clksrc:%d\n"
174  " res-ns:%" PRIu64 "\n"
175  " min-tmo:%" PRIu64 " max-tmo%" PRIu64 "\n"
176  "EM-attr: .res:%" PRIu64 " .max_tmo:%" PRIu64 "",
177  err, odp_clksrc, odp_res_capa.res_ns,
178  odp_res_capa.min_tmo, odp_res_capa.max_tmo,
179  attr.resolution, attr.max_tmo);
180  return EM_TIMER_UNDEF;
181  }
182  /* Store the params */
183  if (attr.resolution == 0)
184  attr.resolution = odp_res_capa.res_ns;
185  if (attr.max_tmo == 0)
186  attr.max_tmo = odp_res_capa.max_tmo;
187  min_tmo = odp_res_capa.min_tmo;
188 
189  /*
190  * Set odp_timer_pool_param_t
191  */
192  odp_timer_pool_param_t odp_tpool_param;
193 
194  memset(&odp_tpool_param, 0, sizeof(odp_timer_pool_param_t));
195 
196  odp_tpool_param.res_ns = attr.resolution;
197  odp_tpool_param.min_tmo = min_tmo;
198  odp_tpool_param.max_tmo = attr.max_tmo;
199  odp_tpool_param.num_timers = attr.num_tmo;
200  odp_tpool_param.priv = attr.flags & EM_TIMER_FLAG_PRIVATE ? 1 : 0;
201  odp_tpool_param.clk_src = odp_clksrc;
202 
203  /*
204  * Set odp_pool_param_t for the EM-timer tmo_pool
205  */
206  odp_pool_param_t odp_pool_param;
207 
208  odp_pool_param_init(&odp_pool_param);
209  odp_pool_param.type = ODP_POOL_BUFFER;
210  odp_pool_param.buf.num = attr.num_tmo;
211  odp_pool_param.buf.size = sizeof(em_timer_timeout_t);
212  odp_pool_param.buf.align = ODP_CACHE_LINE_SIZE;
213 
214  /*
215  * Find a free timer-slot.
216  * This slow search should not be a problem with only a few timers
217  * especially when these are normally created at startup.
218  */
219  event_timer_t *timer;
220  int i;
221 
222  odp_ticketlock_lock(&em_shm->timers.timer_lock);
223 
224  for (i = 0; i < EM_ODP_MAX_TIMERS; i++) {
225  timer = &em_shm->timers.timer[i];
226 
227  if (timer->odp_tmr_pool != ODP_TIMER_POOL_INVALID)
228  continue;
229 
230  char timer_pool_name[ODP_TIMER_POOL_NAME_LEN];
231  char tmo_pool_name[ODP_POOL_NAME_LEN];
232  char *name;
233 
234  if (attr.name[0] != '\0') {
235  name = attr.name;
236  } else {
237  snprintf(timer_pool_name, ODP_TIMER_POOL_NAME_LEN,
238  "EM-timer-%d", timer->idx);
239  name = timer_pool_name;
240  }
241  timer->odp_tmr_pool = odp_timer_pool_create(name,
242  &odp_tpool_param);
243  if (timer->odp_tmr_pool == ODP_TIMER_POOL_INVALID)
244  goto error_locked;
245  TMR_DBG_PRINT("%s(): Created timer: %s with idx: %d\n",
246  __func__, name, timer->idx);
247 
248  snprintf(tmo_pool_name, ODP_POOL_NAME_LEN, "Tmo-pool-%d",
249  timer->idx);
250  timer->tmo_pool = odp_pool_create(tmo_pool_name,
251  &odp_pool_param);
252  if (timer->tmo_pool == ODP_POOL_INVALID)
253  goto error_locked;
254  TMR_DBG_PRINT("%s(): Created ODP-pool: %s for %d timeouts\n",
255  __func__, tmo_pool_name, odp_pool_param.buf.num);
256 
257  timer->flags = attr.flags;
258 
259  odp_timer_pool_start();
260  break;
261  }
262 
263  odp_ticketlock_unlock(&em_shm->timers.timer_lock);
264 
265  return i < EM_ODP_MAX_TIMERS ? (em_timer_t)TMR_I2H(i) : EM_TIMER_UNDEF;
266 
267 error_locked:
268  /* odp_ticketlock_lock(&timer_shm->tlock) */
269 
270  /* 'timer' set in loop */
271  if (timer->tmo_pool != ODP_POOL_INVALID)
272  odp_pool_destroy(timer->tmo_pool);
273  if (timer->odp_tmr_pool != ODP_TIMER_POOL_INVALID)
274  odp_timer_pool_destroy(timer->odp_tmr_pool);
275  timer->tmo_pool = ODP_POOL_INVALID;
276  timer->odp_tmr_pool = ODP_TIMER_POOL_INVALID;
277 
278  odp_ticketlock_unlock(&em_shm->timers.timer_lock);
279 
280  INTERNAL_ERROR(EM_ERR_LIB_FAILED,
281  EM_ESCOPE_TIMER_CREATE,
282  "Timer pool create failed");
283  return EM_TIMER_UNDEF;
284 }

286 em_status_t em_timer_delete(em_timer_t tmr)
287 {
288  timer_storage_t *const tmrs = &em_shm->timers;
289 
290  int i = TMR_H2I(tmr);
291 
292  odp_ticketlock_lock(&tmrs->timer_lock);
293 
294  if (unlikely(!is_timer_valid(tmr))) {
295  odp_ticketlock_unlock(&tmrs->timer_lock);
296  return INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TIMER_DELETE,
297  "Invalid timer:%" PRI_TMR "", tmr);
298  }
299 
300  odp_pool_destroy(tmrs->timer[i].tmo_pool);
301  tmrs->timer[i].tmo_pool = ODP_POOL_INVALID;
302  odp_timer_pool_destroy(tmrs->timer[i].odp_tmr_pool);
303  tmrs->timer[i].odp_tmr_pool = ODP_TIMER_POOL_INVALID;
304 
305  odp_ticketlock_unlock(&tmrs->timer_lock);
306 
307  return EM_OK;
308 }

311 {
312  timer_storage_t *const tmrs = &em_shm->timers;
313 
314  int i = TMR_H2I(tmr);
315 
316  if (EM_CHECK_LEVEL > 0) {

323 em_tmo_t em_tmo_create(em_timer_t tmr, em_tmo_flag_t flags, em_queue_t queue)
324 {
325  int i = TMR_H2I(tmr);
326  odp_timer_pool_t odptmr;
327  queue_elem_t *const q_elem = queue_elem_get(queue);
328  odp_buffer_t tmo_buf;
329 
330  if (EM_CHECK_LEVEL > 0) {
331  if (!is_timer_valid(tmr)) {
332  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TMO_CREATE,
333  "Invalid timer:%" PRI_TMR "", tmr);
334  return EM_TMO_UNDEF;
335  }
336  if (q_elem == NULL || !queue_allocated(q_elem)) {
337  INTERNAL_ERROR(EM_ERR_BAD_ID, EM_ESCOPE_TMO_CREATE,
338  "Tmr:%" PRI_TMR ": inv Q:%" PRI_QUEUE "",
339  tmr, queue);
340  return EM_TMO_UNDEF;
341  }
342  }
343 
344  tmo_buf = odp_buffer_alloc(em_shm->timers.timer[i].tmo_pool);
345 
346  if (unlikely(tmo_buf == ODP_BUFFER_INVALID)) {
347  INTERNAL_ERROR(EM_ERR_ALLOC_FAILED, EM_ESCOPE_TMO_CREATE,
348  "Tmr:%" PRI_TMR ": tmo pool exhausted", tmr);
349  return EM_TMO_UNDEF;
350  }
351 
352  em_timer_timeout_t *tmo = odp_buffer_addr(tmo_buf);
353 
354  odptmr = em_shm->timers.timer[i].odp_tmr_pool;
355  tmo->odp_timer = odp_timer_alloc(odptmr, q_elem->odp_queue, NULL);
356 
357  if (unlikely(tmo->odp_timer == ODP_TIMER_INVALID)) {
358  INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TMO_CREATE,
359  "Tmr:%" PRI_TMR ": odp_timer_alloc() failed",
360  tmr);
361  return EM_TMO_UNDEF;
362  }
363 
364  /* OK, init state */
365  odp_atomic_init_u32(&tmo->state, EM_TMO_STATE_IDLE);
366  tmo->period = 0;
367  tmo->odp_timer_pool = odptmr;
368  tmo->odp_buffer = tmo_buf;
369  tmo->flags = flags;
370  tmo->queue = queue;
371  TMR_DBG_PRINT("%s: ODP tmo %ld\n", __func__,
372  (unsigned long)tmo->odp_timer);
373  return tmo;
374 }

376 em_status_t em_tmo_delete(em_tmo_t tmo, em_event_t *cur_event)
377 {
378  if (EM_CHECK_LEVEL > 0) {
379  RETURN_ERROR_IF(tmo == EM_TMO_UNDEF, EM_ERR_BAD_ID,
380  EM_ESCOPE_TMO_DELETE, "Invalid tmo");
381  RETURN_ERROR_IF(cur_event == NULL, EM_ERR_BAD_POINTER,
382  EM_ESCOPE_TMO_DELETE, "NULL pointer");
383  }
384 
385  if (EM_CHECK_LEVEL > 1) {
386  em_tmo_state_t tmo_state = odp_atomic_load_acq_u32(&tmo->state);
387 
388  RETURN_ERROR_IF(tmo_state == EM_TMO_STATE_UNKNOWN,
389  EM_ERR_BAD_STATE, EM_ESCOPE_TMO_DELETE,
390  "Invalid tmo state:%d", tmo_state);
391  RETURN_ERROR_IF(!odp_buffer_is_valid(tmo->odp_buffer),
392  EM_ERR_BAD_ID, EM_ESCOPE_TMO_DELETE,
393  "Invalid tmo buffer");
394  }
395 
396  odp_event_t odp_evt = odp_timer_free(tmo->odp_timer);
397 
398  if (odp_evt != ODP_EVENT_INVALID)
399  *cur_event = event_odp2em(odp_evt);
400  else
401  *cur_event = EM_EVENT_UNDEF;
402 
403  odp_buffer_t tmp = tmo->odp_buffer;
404 
405  tmo->odp_timer = ODP_TIMER_INVALID;
406  odp_atomic_store_rel_u32(&tmo->state, EM_TMO_STATE_UNKNOWN);
407  tmo->odp_buffer = ODP_BUFFER_INVALID;
408  odp_buffer_free(tmp);
409 
410  return EM_OK;
411 }

413 em_status_t em_tmo_set_abs(em_tmo_t tmo, em_timer_tick_t ticks_abs,
414  em_event_t tmo_ev)
415 {
416  if (EM_CHECK_LEVEL > 0) {
417  RETURN_ERROR_IF(tmo == EM_TMO_UNDEF || tmo_ev == EM_EVENT_UNDEF,
418  EM_ERR_BAD_ID, EM_ESCOPE_TMO_SET_ABS,
419  "Inv.args: tmo:%" PRI_TMO " ev:%" PRI_EVENT "",
420  tmo, tmo_ev);
421  }
422 
423  if (EM_CHECK_LEVEL > 0) {
424  RETURN_ERROR_IF(tmo->flags & EM_TMO_FLAG_PERIODIC,
425  EM_ERR_BAD_CONTEXT, EM_ESCOPE_TMO_SET_ABS,
426  "Periodic tmo flag set");
427  }
428  if (EM_CHECK_LEVEL > 1) {
429  em_tmo_state_t tmo_state = odp_atomic_load_acq_u32(&tmo->state);
430 
431  RETURN_ERROR_IF(tmo_state == EM_TMO_STATE_UNKNOWN,
432  EM_ERR_BAD_STATE, EM_ESCOPE_TMO_SET_ABS,
433  "Invalid tmo state:%d", tmo_state);
434  RETURN_ERROR_IF(!odp_buffer_is_valid(tmo->odp_buffer),
435  EM_ERR_BAD_ID, EM_ESCOPE_TMO_SET_ABS,
436  "Invalid tmo buffer");
437  }
438 
439  odp_event_t odp_ev = event_em2odp(tmo_ev);
440  /* set tmo active and arm with absolute time */
441  odp_atomic_store_rel_u32(&tmo->state, EM_TMO_STATE_ACTIVE);
442  int ret = odp_timer_set_abs(tmo->odp_timer, ticks_abs, &odp_ev);
443 
444  TMR_DBG_PRINT("%s(): ODP ret %d\n", __func__, ret);
445 
446  if (unlikely(ret != ODP_TIMER_SUCCESS)) {
447  odp_atomic_store_rel_u32(&tmo->state, EM_TMO_STATE_IDLE);
448  return INTERNAL_ERROR(EM_ERR_LIB_FAILED, EM_ESCOPE_TMO_SET_ABS,
449  "odp_timer_set_abs():%d", ret);
450  }
451 
452  return EM_OK;
453 }

495 em_status_t em_tmo_cancel(em_tmo_t tmo, em_event_t *cur_event)
496 {
497  if (EM_CHECK_LEVEL > 0) {
498  RETURN_ERROR_IF(tmo == EM_TMO_UNDEF, EM_ERR_BAD_ID,
499  EM_ESCOPE_TMO_CANCEL, "Invalid tmo");
500  RETURN_ERROR_IF(cur_event == NULL, EM_ERR_BAD_POINTER,
501  EM_ESCOPE_TMO_CANCEL, "NULL pointer");
502  }
503 
504  if (EM_CHECK_LEVEL > 1) {
505  em_tmo_state_t tmo_state = odp_atomic_load_acq_u32(&tmo->state);
506 
507  RETURN_ERROR_IF(tmo_state == EM_TMO_STATE_UNKNOWN,
508  EM_ERR_BAD_STATE, EM_ESCOPE_TMO_CANCEL,
509  "Invalid tmo state:%d", tmo_state);
510  RETURN_ERROR_IF(!odp_buffer_is_valid(tmo->odp_buffer),
511  EM_ERR_BAD_ID, EM_ESCOPE_TMO_CANCEL,
512  "Invalid tmo buffer");
513  }
514 
515  /* cancel and set tmo idle */
516  odp_event_t odp_ev = ODP_EVENT_INVALID;
517  int ret = odp_timer_cancel(tmo->odp_timer, &odp_ev);
518 
519  odp_atomic_store_rel_u32(&tmo->state, EM_TMO_STATE_IDLE);
520  if (ret == 0) {
521  *cur_event = event_odp2em(odp_ev);
522  return EM_OK;

529 em_status_t em_tmo_ack(em_tmo_t tmo, em_event_t next_tmo_ev)
530 {
531  if (EM_CHECK_LEVEL > 0) {
532  RETURN_ERROR_IF(tmo == EM_TMO_UNDEF ||
533  next_tmo_ev == EM_EVENT_UNDEF,
534  EM_ERR_BAD_ID, EM_ESCOPE_TMO_ACK,
535  "Inv.args: tmo:%" PRI_TMO " ev:%" PRI_EVENT "",
536  tmo, next_tmo_ev);
537  RETURN_ERROR_IF(!(tmo->flags & EM_TMO_FLAG_PERIODIC),
538  EM_ERR_BAD_CONTEXT, EM_ESCOPE_TMO_ACK,
539  "Tmo ACK: Not a periodic tmo");
540  }
541 
542  em_tmo_state_t tmo_state = odp_atomic_load_acq_u32(&tmo->state);
543  /*
544  * If tmo cancelled:
545  * Return an error so the application can free the given event.
546  */
547  RETURN_ERROR_IF(tmo_state != EM_TMO_STATE_ACTIVE,
548  EM_ERR_BAD_STATE, EM_ESCOPE_TMO_ACK,
549  "Tmo ACK: invalid tmo state:%d", tmo_state);
550 
551  if (EM_CHECK_LEVEL > 1) {
552  RETURN_ERROR_IF(!odp_buffer_is_valid(tmo->odp_buffer),
553  EM_ERR_BAD_ID, EM_ESCOPE_TMO_ACK,
554  "Tmo ACK: invalid tmo buffer");
555  }
556 
557  /*
558  * The periodic timer will silently stop if ack fails! Attempt to
559  * handle exceptions and, if the tmo cannot be renewed, call
560  * the errorhandler so the application may recover.
561  */
562  tmo->last_tick += tmo->period; /* maintain absolute time */
563  odp_event_t odp_ev = event_em2odp(next_tmo_ev);
564  int ret;
565 
566  do {
567  /* ask new timeout for next period */
568  ret = odp_timer_set_abs(tmo->odp_timer,
569  tmo->last_tick + tmo->period,
570  &odp_ev);
571  /*
572  * Calling ack() was delayed over one period if 'ret' is
573  * ODP_TIMER_TOOEARLY, i.e. now in past. Other errors
574  * should not happen.
575  */
576  if (likely(ret != ODP_TIMER_TOOEARLY))
577  break;
578 
579  /* ack() delayed beyond next time slot */
580  if (tmo->flags & EM_TMO_FLAG_NOSKIP) {
581  /* can't skip, send immediately */
582  em_status_t ret = em_send(next_tmo_ev, tmo->queue);
583 
584  RETURN_ERROR_IF(ret != EM_OK, EM_ERR_OPERATION_FAILED,
585  EM_ESCOPE_TMO_ACK,
586  "Tmo ACK: failed to send");
587  return EM_OK;
588  }
589 
590  do { /* catch up time */
591  tmo->last_tick += tmo->period;
592  } while (tmo->last_tick + tmo->period <
593  odp_timer_current_tick(tmo->odp_timer_pool));
594  /* if this takes too much time we can hit TOOEARLY again */
595 
596  } while (ret != ODP_TIMER_SUCCESS);
597 
598  RETURN_ERROR_IF(ret != ODP_TIMER_SUCCESS, EM_ERR_LIB_FAILED,
599  EM_ESCOPE_TMO_ACK, "Tmo ACK: failed to renew");
600  return EM_OK;
601 }

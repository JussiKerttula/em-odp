44 /**
45  * Helper function for running all dispatcher enter callback functions.
46  *
47  * Note: All callbacks are run even if the event is freed (=EM_EVENT_UNDEF).
48  *
49  * @param eo EO handle
50  * @param eo_ctx EO context data
51  * @param event Event handle
52  * @param type Event type
53  * @param queue Queue from which this event came from
54  * @param q_ctx Queue context data
55  */
56 static inline void
57 dispatch_enter_cb(em_eo_t eo, void **eo_ctx, em_event_t *event,
58  em_event_type_t *type, em_queue_t *queue, void **q_ctx)
59 {
60  hook_tbl_t *const dispatch_enter_cb_tbl = em_shm->dispatch_enter_cb_tbl;
61  em_dispatch_enter_func_t dispatch_enter_fn;
62  int i;
63 
64  for (i = 0; i < EM_CALLBACKS_MAX; i++) {
65  dispatch_enter_fn = dispatch_enter_cb_tbl->tbl[i].disp_enter;
66  if (dispatch_enter_fn == NULL)
67  return;
68  dispatch_enter_fn(eo, eo_ctx, event, type, queue, q_ctx);
69  }
70 }

92 static inline void
93 call_eo_receive_fn(const em_eo_t eo, const em_receive_func_t eo_receive_func,
94  event_hdr_t *const ev_hdr, queue_elem_t *const q_elem)
95 {
96  em_queue_t queue = q_elem->queue;
97  void *queue_ctx = q_elem->context;
98  void *eo_ctx = q_elem->eo_ctx;
99 
100  em_event_t event = event_hdr_to_event(ev_hdr);
101  em_event_type_t event_type = ev_hdr->event_type;
102 
103  if (EM_CHECK_LEVEL > 2 &&
104  unlikely(env_atomic32_get(&ev_hdr->allocated) != 1)) {
105  const char *eo_name = q_elem->eo_elem == NULL ?
106  "" : q_elem->eo_elem->name;
107  INTERNAL_ERROR(EM_FATAL(EM_ERR_BAD_STATE), EM_ESCOPE_DISPATCH,
108  "EO:%" PRI_EO ":%s Q:%" PRI_QUEUE "\n"
109  "Event:%" PRI_EVENT " already freed",
110  eo, eo_name, queue, event);
111  }
112 
113  em_locm.current.q_elem = q_elem;
114  /* Check and set core local event group (before dispatch callback(s)) */
115  event_group_set_local(ev_hdr);
116 
117  if (EM_DISPATCH_CALLBACKS_ENABLE)
118  dispatch_enter_cb(eo, &eo_ctx, &event, &event_type,
119  &queue, &queue_ctx);
120 
121  if (likely(event != EM_EVENT_UNDEF)) {
122  /*
123  * Call the EO receive function
124  * (only if the dispatch callback(s) did not free the event)
125  */
126  eo_receive_func(eo_ctx, event, event_type,
127  queue, queue_ctx);
128  }
129 
130  if (EM_DISPATCH_CALLBACKS_ENABLE)
131  dispatch_exit_cb(eo);
132 
133  /*
134  * Event belongs to an event_group, update the count and
135  * if requested send notifications
136  */
137  if (em_locm.current.egrp != EM_EVENT_GROUP_UNDEF) {
138  /*
139  * Atomically decrease the event group count.
140  * If the new count is zero, send notification events.
141  */
142  event_group_count_decrement(1);
143  }
144  em_locm.current.egrp = EM_EVENT_GROUP_UNDEF;
145 }

147 static inline void
148 check_local_queues(void)
149 {
150  event_hdr_t *ev_hdr;
151 
152  /*
153  * Check if the previous EO receive function sent events to a
154  * local queue ('EM_QUEUE_TYPE_LOCAL') - and if so, dispatch
155  * those events immediately.
156  */
157  while ((ev_hdr = next_local_queue_event()) != NULL) {
158  /* Dispatch event from a local queue */
159  queue_elem_t *const local_q_elem = ev_hdr->q_elem;
160  const em_eo_t eo = local_q_elem->eo;
161  const em_receive_func_t eo_receive_func =
162  local_q_elem->receive_func;
163 
164  /*
165  * Call the Execution Object (EO) receive function.
166  * Scheduling context may be released during this
167  */
168  call_eo_receive_fn(eo, eo_receive_func, ev_hdr, local_q_elem);
169  }
170 }

172 /**
173  * Dispatch events - call the EO-receive functions and pass the
174  * events for processing
175  */
176 static inline void
177 dispatch_events(event_hdr_t *const ev_hdr_tbl[], const int num_events,
178  queue_elem_t *const q_elem)
179 {
180  const em_queue_type_t q_type = q_elem->type;
181  const em_eo_t eo = q_elem->eo;
182  const em_receive_func_t eo_receive_func = q_elem->receive_func;
183  em_sched_context_type_t sched_ctx_type = EM_SCHED_CONTEXT_TYPE_NONE;
184 
185  if (q_type == EM_QUEUE_TYPE_ATOMIC)
186  sched_ctx_type = EM_SCHED_CONTEXT_TYPE_ATOMIC;
187  else if (q_type == EM_QUEUE_TYPE_PARALLEL_ORDERED)
188  sched_ctx_type = EM_SCHED_CONTEXT_TYPE_ORDERED;
189 
190  em_locm.current.sched_context_type = sched_ctx_type;
191  em_locm.current.sched_q_elem = q_elem;
192  /* here: em_locm.current.egrp == EM_EVENT_GROUP_UNDEF */
193 
194  /*
195  * Call the Execution Object (EO) receive function for each event.
196  * Scheduling context may be released during this.
197  */
198  for (int i = 0; i < num_events; i++) {
199  event_hdr_t *const ev_hdr = ev_hdr_tbl[i];
200 
201  em_locm.event_burst_cnt--;
202  call_eo_receive_fn(eo, eo_receive_func, ev_hdr, q_elem);
203  check_local_queues();
204  }
205 
206  /*
207  * Check for buffered events sent to output queues during the previous
208  * dispatch rounds
209  */
210  if (!EM_OUTPUT_QUEUE_IMMEDIATE &&
211  em_locm.output_queue_track.idx_cnt > 0)
212  output_queue_buffering_drain();
213 
214  em_locm.current.q_elem = NULL;
215  em_locm.current.sched_q_elem = NULL;
216  em_locm.current.sched_context_type = EM_SCHED_CONTEXT_TYPE_NONE;
217 }

219 static inline void
220 dispatch_local_queues(void)
221 {
222  event_hdr_t *const ev_hdr = next_local_queue_event();
223  queue_elem_t *local_q_elem;
224 
225  if (ev_hdr != NULL) {
226  local_q_elem = ev_hdr->q_elem;
227  dispatch_events(&ev_hdr, 1, local_q_elem);
228  }
229 }

231 /*
232  * Run a dispatch round - query the scheduler for events and dispatch
233  */
234 static inline int
235 dispatch_round(void)
236 {
237  odp_queue_t odp_queue;
238  odp_event_t odp_ev_tbl[EM_SCHED_MULTI_MAX_BURST];
239  event_hdr_t *ev_hdr_tbl[EM_SCHED_MULTI_MAX_BURST];
240  em_event_t *ev_tbl;
241  queue_elem_t *queue_elem;
242 
243  /*
244  * Schedule events to the core from queues
245  */
246  const int num_events =
247  odp_schedule_multi_no_wait(&odp_queue, odp_ev_tbl,
248  EM_SCHED_MULTI_MAX_BURST);
249  if (unlikely(num_events <= 0)) {
250  /*
251  * No scheduled events available, check if the local queues
252  * contain anything on this core - e.g. pktio or something
253  * outside the dispatch-context might have sent to a local queue
254  */
255  dispatch_local_queues();
256  return 0;
257  }
258 
259  ev_tbl = events_odp2em(odp_ev_tbl);
260  /* Events might originate from outside of EM and need hdr-init */
261  event_to_hdr_init_multi(ev_tbl, ev_hdr_tbl, num_events);
262 
263  queue_elem = odp_queue_context(odp_queue);
264  if (unlikely(queue_elem == NULL ||
265  queue_elem->state != EM_QUEUE_STATE_READY)) {
266  /* Drop all events dequeued from this queue */
267  event_free_multi(ev_tbl, num_events);
268 
269  if (queue_elem == NULL)
270  INTERNAL_ERROR(EM_ERR_BAD_POINTER, EM_ESCOPE_DISPATCH,
271  "Event(s) from non-EM Q, drop %d events",
272  num_events);
273  else
274  INTERNAL_ERROR(EM_ERR_BAD_STATE, EM_ESCOPE_DISPATCH,
275  "Q:%" PRI_QUEUE " not ready, state=%d\n"
276  " drop:%d event(s)\n",
277  queue_elem->queue, queue_elem->state,
278  num_events);
279  return 0;
280  }
281 
282  if (queue_elem->atomic_group == EM_ATOMIC_GROUP_UNDEF) {
283  em_locm.event_burst_cnt = num_events;
284  dispatch_events(ev_hdr_tbl, num_events, queue_elem);
285  } else {
286  atomic_group_dispatch(ev_tbl, ev_hdr_tbl,
287  num_events, queue_elem);
288  }
289 
290  return num_events;
291 }

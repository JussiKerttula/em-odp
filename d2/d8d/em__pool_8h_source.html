95 static inline int
96 pool_find_subpool(mpool_elem_t *const pool_elem, size_t size)
97 {
98  int subpool = 0;
99 
100  if (EM_MAX_SUBPOOLS > 1) { /* Compile time option */
101  int i;
102  /* Find the optimal subpool to allocate the event from */
103  for (i = 0; i < pool_elem->num_subpools &&
104  size > pool_elem->size[i]; i++)
105  ;
106 
107  if (unlikely(i >= pool_elem->num_subpools))
108  return -1;
109 
110  subpool = i;
111  }
112 
113  return subpool;
114 }

34 /**
35  * @file
36  * @defgroup em_event Events
37  * Operations on an event.
38  * @{
39  *
40  * All application processing is driven by events in the Event Machine. An event
41  * describes a piece of work. The structure of an event is implementation and
42  * event type specific: it may be a directly accessible buffer of memory, a
43  * descriptor containing a list of buffer pointers, a descriptor of a packet
44  * buffer etc.
45  *
46  * Applications use the event type to interpret the event structure.
47  *
48  * Events follow message passing semantics: an event has to be allocated using
49  * the provided API (em_alloc()) or received through queues by an EO callback
50  * function after which the event is owned by the application. Event ownership
51  * is transferred back to the system by using em_send() or em_free().
52  * An event not owned by the application should not be touched.
53  *
54  * Since em_event_t may not carry a direct pointer value to the event structure,
55  * em_event_pointer() must be used to translate an event to an event structure
56  * pointer (for maintaining portability).
57  *
58  * em_event_t is defined in event_machine_types.h
59  *
60  * @see em_event_pointer()
61  */

70 /**
71  * Allocate an event.
72  *
73  * The memory address of the allocated event is system specific and can depend
74  * on the given pool id, the event size and type. The returned event (handle)
75  * may refer to a memory buffer or a HW specific descriptor, i.e. the event
76  * structure is system specific.
77  *
78  * Use em_event_pointer() to convert an event (handle) to a pointer to an event
79  * structure.
80  *
81  * EM_EVENT_TYPE_SW with minor type '0' is reserved for direct portability.
82  * It is always guaranteed to return a 64-bit aligned contiguous data buffer,
83  * that can directly be used by the application up to the given size (no HW
84  * specific descriptors etc. are visible).
85  *
86  * EM_POOL_DEFAULT can be used as pool handle if there's no need to use a
87  * specific memory pool.
88  *
89  * Additionally it is guaranteed, that two separate buffers never share a cache
90  * line (to avoid false sharing).
91  *
92  * @param size Event size in octets (size > 0)
93  * @param type Event type to allocate
94  * @param pool Event pool handle
95  *
96  * @return The allocated event or EM_EVENT_UNDEF on error.
97  *
98  * @see em_free(), em_send(), em_event_pointer(), em_receive_func_t()
99  */
100 em_event_t em_alloc(size_t size, em_event_type_t type, em_pool_t pool);
101 
102 /**
103  * Free an event.
104  *
105  * It is assumed that the implementation can detect the memory area/pool that
106  * the event was originally allocated from.
107  *
108  * The em_free() function transfers ownership of the event back to the system
109  * and the application must not touch the event (or related memory buffers)
110  * after calling it.
111  *
112  * The application must only free events it owns. For example, the sender must
113  * not free an event after sending it.
114  *
115  * @param event Event to be freed
116  *
117  * @see em_alloc(), em_receive_func_t()
118  */
119 void em_free(em_event_t event);
120 
121 /**
122  * Send an event to a queue.
123  *
124  * The event must have been allocated with em_alloc(), or received via an EO
125  * receive-function. The sender must not touch the event after calling em_send()
126  * as the ownership has been transferred to the system or possibly to the next
127  * receiver. If the return status is *not* EM_OK, the ownership has not been
128  * transferred and the application is still responsible for the event (e.g. may
129  * free it).
130  *
131  * EM does not currently define guaranteed event delivery, i.e. EM_OK return
132  * value only means the event was accepted for delivery. It could still be lost
133  * during delivery (e.g. due to a removed queue or system congestion, etc).
134  *
135  * @param event Event to be sent
136  * @param queue Destination queue
137  *
138  * @return EM_OK if successful (accepted for delivery).
139  *
140  * @see em_alloc()
141  */
142 em_status_t em_send(em_event_t event, em_queue_t queue);
143 
144 /**
145  * Send multiple events to a queue.
146  *
147  * As em_send, but multiple events can be sent with one call for potential
148  * performance gain.
149  * The function returns the number of events actually sent. A return value equal
150  * to the given 'num' means that all events were sent. A return value less than
151  * 'num' means that only the first 'num' events were sent and the rest must be
152  * handled by the application.
153  *
154  * @param events List of events to be sent (i.e. ptr to array of events)
155  * @param num Number of events (positive integer)
156  * @param queue Destination queue
157  *
158  * @return number of events successfully sent (equal to num if all successful)
159  *
160  * @see em_send()
161  */
162 int em_send_multi(em_event_t *const events, int num, em_queue_t queue);
